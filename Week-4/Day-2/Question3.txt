class MyHashSet {
public:
    int arr[1000001];
    MyHashSet() {
        memset(arr, 0, sizeof(int)*1000001);
    }
    
    void add(int key) {
        arr[key]=1;
    }
    
    void remove(int key) {
        arr[key]=0;
    }
    
    /** Returns true if this set contains the specified element */
    bool contains(int key) {
        if(arr[key]==1){return true;}
        return false;
    }
};
