class Solution {
public:
    vector<int> singleNumber(vector<int>& nums) {
        map<int, int> h;
        vector<int> ans;
        for(int i=0;i<nums.size();i++){
            h[nums[i]]++;
        }
        for(auto i:h){
            if(i.second==1){
                ans.push_back(i.first);
            }
        }
        return ans;
    }
};
